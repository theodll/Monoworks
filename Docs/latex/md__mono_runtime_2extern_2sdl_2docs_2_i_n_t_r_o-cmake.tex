The easiest way to use SDL is to include it as a subproject in your project.

We\textquotesingle{}ll start by creating a simple project to build and run \href{hello.c}{\texttt{hello.\+c}}\hypertarget{md__mono_runtime_2extern_2sdl_2docs_2_i_n_t_r_o-cmake_autotoc_md1165}{}\doxysection{\texorpdfstring{Get a copy of the SDL source:\+}{Get a copy of the SDL source:\+}}\label{md__mono_runtime_2extern_2sdl_2docs_2_i_n_t_r_o-cmake_autotoc_md1165}

\begin{DoxyCode}{0}
\DoxyCodeLine{git\ clone\ https://github.com/libsdl-\/org/SDL.git\ vendored/SDL}

\end{DoxyCode}
\hypertarget{md__mono_runtime_2extern_2sdl_2docs_2_i_n_t_r_o-cmake_autotoc_md1166}{}\doxysection{\texorpdfstring{Create the file CMake\+Lists.\+txt}{Create the file CMake\+Lists.\+txt}}\label{md__mono_runtime_2extern_2sdl_2docs_2_i_n_t_r_o-cmake_autotoc_md1166}

\begin{DoxyCode}{0}
\DoxyCodeLine{cmake\_minimum\_required(VERSION\ 3.16)}
\DoxyCodeLine{project(hello)}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ set\ the\ output\ directory\ for\ built\ objects.}
\DoxyCodeLine{\#\ This\ makes\ sure\ that\ the\ dynamic\ library\ goes\ into\ the\ build\ directory\ automatically.}
\DoxyCodeLine{set(CMAKE\_RUNTIME\_OUTPUT\_DIRECTORY\ "{}\$\{CMAKE\_BINARY\_DIR\}/\$<CONFIGURATION>"{})}
\DoxyCodeLine{set(CMAKE\_LIBRARY\_OUTPUT\_DIRECTORY\ "{}\$\{CMAKE\_BINARY\_DIR\}/\$<CONFIGURATION>"{})}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ This\ assumes\ the\ SDL\ source\ is\ available\ in\ vendored/SDL}
\DoxyCodeLine{add\_subdirectory(vendored/SDL\ EXCLUDE\_FROM\_ALL)}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ Create\ your\ game\ executable\ target\ as\ usual}
\DoxyCodeLine{add\_executable(hello\ WIN32\ hello.c)}
\DoxyCodeLine{}
\DoxyCodeLine{\#\ Link\ to\ the\ actual\ SDL3\ library.}
\DoxyCodeLine{target\_link\_libraries(hello\ PRIVATE\ SDL3::SDL3)}

\end{DoxyCode}
\hypertarget{md__mono_runtime_2extern_2sdl_2docs_2_i_n_t_r_o-cmake_autotoc_md1167}{}\doxysection{\texorpdfstring{Configure and Build:\+}{Configure and Build:\+}}\label{md__mono_runtime_2extern_2sdl_2docs_2_i_n_t_r_o-cmake_autotoc_md1167}

\begin{DoxyCode}{0}
\DoxyCodeLine{cmake\ -\/S\ .\ -\/B\ build}
\DoxyCodeLine{cmake\ -\/-\/build\ build}

\end{DoxyCode}
\hypertarget{md__mono_runtime_2extern_2sdl_2docs_2_i_n_t_r_o-cmake_autotoc_md1168}{}\doxysection{\texorpdfstring{Run:\+}{Run:\+}}\label{md__mono_runtime_2extern_2sdl_2docs_2_i_n_t_r_o-cmake_autotoc_md1168}
The executable should be in the {\ttfamily \doxylink{namespacebuild}{build}} directory:\+


\begin{DoxyCode}{0}
\DoxyCodeLine{cd\ build}
\DoxyCodeLine{./hello}

\end{DoxyCode}


If there wasn\textquotesingle{}t an executable there despite the above Build section running successfully, it\textquotesingle{}s likely because you\textquotesingle{}re following this guide using the Visual Studio toolchain, it should instead be in the {\ttfamily build/\+\+Debug} directory:\+ 
\begin{DoxyCode}{0}
\DoxyCodeLine{cd\ build/Debug}
\DoxyCodeLine{./hello}

\end{DoxyCode}
